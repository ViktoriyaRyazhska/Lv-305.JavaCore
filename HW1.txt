package ht305;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class FHT {
	public static void main(String[] args) throws NumberFormatException, IOException
	{
		BufferedReader br = new BufferedReader(
				new InputStreamReader(System.in));
		System.out.println("Input radius: ");
		double R = Double.parseDouble(br.readLine());
		double per = 2 * Math.PI * R;
		double area = Math.PI * R * R;
		System.out.println("Perimetr = " + per);
		System.out.println("Area = " + area);
	}
}

package ht305;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class THT {
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(
				new InputStreamReader(System.in));
		System.out.println("Input call price to France: ");
		double c1 = Double.parseDouble(br.readLine());
		System.out.println("Input call price to Canada: ");
		double c2 = Double.parseDouble(br.readLine());
		System.out.println("Input call price to Poland: ");
		double c3 = Double.parseDouble(br.readLine());
		System.out.println("Call to Canada costs " + c2 + "hr/min");
		System.out.println("Call to France costs " + c1 + "hr/min");
		System.out.println("Call to Poland costs " + c3 + "hr/min");
		System.out.println("How long was your call to France? ");
		double t1 = Double.parseDouble(br.readLine());
		System.out.println("How long was your call to Canada? ");
		double t2 = Double.parseDouble(br.readLine());
		System.out.println("How long was your call to Poland? ");
		double t3 = Double.parseDouble(br.readLine());
		double totalTime = t1 + t2 + t3;
		System.out.println("Call to Canada continued " + t2 + "minutes");
		System.out.println("Call to France continued " + t1 + "minutes");
		System.out.println("Call to Poland continued " + t3 + "minutes");
		double cost1 = c1 * t1;
		double cost2 = c2 * t2;
		double cost3 = c3 * t3;
		double totalCost = cost1 + cost2 + cost3;
		System.out.println("So, your call to Canada continued  " + t2 + "minutes and you spent " + cost2 + "hr");
		System.out.println("So, your call to France continued  " + t1 + "minutes and you spent " + cost1 + "hr");
		System.out.println("So, your call to Poland continued  " + t3 + "minutes and you spent " + cost3 + "hr");
		System.out.println("Totaly, you spent " + totalTime + "min and " + totalCost + "hr");
	}
}
